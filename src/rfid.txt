#include <SPI.h>
#include <MFRC522.h>

// Pin configuration (match your wiring)
constexpr uint8_t RST_PIN  = 22; // RST -> GPIO22
constexpr uint8_t SS_PIN   = 5;  // SDA(SS) -> GPIO5

// On-board LED (some ESP32 dev boards use 2 or 21; common is GPIO2)
constexpr uint8_t LED_PIN = 2;

MFRC522 mfrc522(SS_PIN, RST_PIN); // Create MFRC522 instance

void setup() {
  Serial.begin(115200);
  while (!Serial) { delay(10); } // wait for Serial (optional)

  pinMode(LED_PIN, OUTPUT);
  digitalWrite(LED_PIN, LOW);

  SPI.begin();           // Init SPI bus
  mfrc522.PCD_Init();    // Init MFRC522
  delay(50);

  Serial.println();
  Serial.println("RC522 + ESP32 ready");
  Serial.print("Using SS_PIN: "); Serial.println(SS_PIN);
  Serial.print("Using RST_PIN: "); Serial.println(RST_PIN);
  Serial.println("Scan a card/tag to see UID");
}

void loop() {
  // Look for new cards
  if (!mfrc522.PICC_IsNewCardPresent()) {
    // no card present
    return;
  }

  // Select one of the cards
  if (!mfrc522.PICC_ReadCardSerial()) {
    // Could not read card serial
    return;
  }

  // Card detected and UID read
  digitalWrite(LED_PIN, HIGH);

  // Print UID in HEX
  Serial.print("UID: ");
  String uidHex = "";
  for (byte i = 0; i < mfrc522.uid.size; i++) {
    if (mfrc522.uid.uidByte[i] < 0x10) Serial.print("0");
    Serial.print(mfrc522.uid.uidByte[i], HEX);
    Serial.print(i < mfrc522.uid.size - 1 ? ":" : "");
    // Build hex string
    char buf[3];
    sprintf(buf, "%02X", mfrc522.uid.uidByte[i]);
    uidHex += String(buf);
  }
  Serial.print("  (HEX)  ");

  // Print UID in DEC (concatenated)
  unsigned long uidDec = 0;
  for (byte i = 0; i < mfrc522.uid.size; i++) {
    uidDec = (uidDec << 8) | mfrc522.uid.uidByte[i];
  }
  Serial.print(uidDec);
  Serial.print(" (DEC)  ");

  // Print uidHex as string too
  Serial.print("string: ");
  Serial.println(uidHex);

  // Optional: print more card info
  Serial.print("SAK: 0x"); Serial.println(mfrc522.uid.sak, HEX);
  Serial.print("PICC Type: ");
  MFRC522::PICC_Type piccType = mfrc522.PICC_GetType(mfrc522.uid.sak);
  Serial.println(mfrc522.PICC_GetTypeName(piccType));

  // Halt PICC and stop encryption on PCD
  mfrc522.PICC_HaltA();
  mfrc522.PCD_StopCrypto1();

  // LED blink to show detection
  delay(200);
  digitalWrite(LED_PIN, LOW);

  // brief delay to avoid multiple reads of the same tag
  delay(300);
}
